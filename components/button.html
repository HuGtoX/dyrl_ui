加载状态 ​

vue
<script setup>
const handleClick = async () => {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve();
    }, 3000);
  });
};
</script>

<template>
  <Space size="8">
    <Button :loading="true" type="primary">基本loading</Button>
    <Button :onClick="handleClick" type="primary"> 异步加载loading </Button>
  </Space>
</template>
